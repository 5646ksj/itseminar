function fnMymenuSeacrhBack(data){

	var prePARENT_PAGE_ID="";
	$("#tMenubody").remove();	
	$('<ul id="tMenubody"></ul>').appendTo("#favorList");
	$(".setLocaOption").remove();
	
	for(i = 0 ; i < repaymentCnt; i++){
		var obj=data.msg.servicedata.OPEN_MYMENU[i];
			
			var	innerHtmlString='';
			innerHtmlString+='<li class="'+obj.PAGE_ID+'" id='+i+'>';
			
			if(obj.IS_DIR=="F"){
				innerHtmlString+='<a href=/quics?page='+obj.PAGE_ID+'>'+obj.DIR_NM+'</a>';
			}else{
				innerHtmlString+='<button type="button" onclick="toggleFavorGroup(this)">'+obj.DIR_NM+'</button>';
			}
			innerHtmlString+='</li>';
				
			if(obj.PARENT_PAGE_ID!=""){
				if(obj.PARENT_PAGE_ID=="root" ){
					
					$(innerHtmlString).appendTo("#tMenubody");
				}else{
					
					var aClass="."+obj.PARENT_PAGE_ID;
					
					aClass=$("li."+aClass);
					var SubStr='';
					if(prePARENT_PAGE_ID!=obj.PARENT_PAGE_ID){
						SubStr+='<ul class="'+obj.PARENT_PAGE_ID+'">';
						SubStr+=innerHtmlString;
						SubStr+='</ul>';						
					
						$(SubStr).appendTo(aClass);
					}else{
					
						var bClass="."+obj.PARENT_PAGE_ID;
					
						bClass=$("ul."+bClass);						
						SubStr+=innerHtmlString;
						$(SubStr).appendTo(bClass);
				
					}
					prePARENT_PAGE_ID=obj.PARENT_PAGE_ID;
				}
			}
		// 폴더일때만  select box  append		
		if(obj.IS_DIR=="T"){
				var strPageID=obj.PAGE_ID;
				var strDirNm=obj.DIR_NM;
				if(strDirNm==""){
					strDirNm="홈";
				}
	
				$("#setLoca").append('<option class="setLocaOption" value='+strPageID+'>'+strDirNm+'</option>');
		}
	}

function fnmenuAddBack(data){
	
	if(data.msg.servicedata.RESULT=="false"){
		alert('즐겨찾기등록에 실패하였습니다.');
		return;
	}
	if(data.msg.servicedata.ERROR_CODE!=""){
		if(data.msg.servicedata.ERROR_CODE=="-5001"){
			alert("디렉토리 등록 개수가 초과 초과하였습니다. 50개까지 등록가능합니다.");
		}else if(data.msg.servicedata.ERROR_CODE=="-5002"){
			alert("페이지 등록 개수가 초과하였습니다. 50개까지 등록가능합니다.");
		}else if(data.msg.servicedata.ERROR_CODE=="-5003"){
			alert("현재페이지는 등록된 페이지입니다.");
		}else{ 
			alert('즐겨찾기등록에 실패하였습니다.');
		}
		return;
	}
	
	fnMymenuSeacrhBack(data);	
}
